--------------------------------------------------------------------------------
-- Labels are Strings.

Label : Set
Label = String

--------------------------------------------------------------------------------
-- Kinds.

data Kind : Level → Set where
  ★ : (ℓ : Level) → Kind ℓ
  _`→_ : ∀ {ℓ₁ ℓ₂} → Kind ℓ₁ → Kind ℓ₂ → Kind (ℓ₁ ⊔ ℓ₂)
  L : ∀ {ℓ} → Kind ℓ
  R[_] : ∀ {ℓ} → Kind ℓ → Kind ℓ

-- type synonyms
lone ltwo lthree : Level
lone   = lsuc lzero
ltwo   = lsuc lone
lthree = lsuc ltwo

★₀ = ★ lzero
★₁ = ★ lone
★₂ = ★ ltwo

--------------------------------------------------------------------------------
-- Kinding Environments, types, and predicates.
--
-- Kinding Environments, types, and predicates are tied up together, like so:
--   - Pred references Ty, KEnv
--   - Ty   references KEnv
--   - KEnv references Pred

data KEnv : Level → Set
data Ty : {ℓ ι : Level} → KEnv ℓ → Kind ι →  Set
data Pred {ℓ ι : Level} (Δ : KEnv ℓ) (κ : Kind ι) : Set

data Pred Δ κ where
  _≲_ : (ρ₁ : Ty Δ R[ κ ]) →
        (ρ₂ : Ty Δ R[ κ ]) →
        Pred Δ κ

  _·_~_ : (ρ₁ : Ty Δ R[ κ ]) →
          (ρ₂ : Ty Δ R[ κ ]) →
          (ρ₃ : Ty Δ R[ κ ]) →
          Pred Δ κ

data KEnv where
  ε    : KEnv lzero
  _,_  : ∀ {ℓ ι} → KEnv ℓ → Kind ι → KEnv (ℓ ⊔ ι)

--------------------------------------------------------------------------------
-- Type vars.
data TVar : ∀ {ℓ ι} → KEnv ℓ → Kind ι → Set where
  Z : ∀ {ℓ₁ ℓ₂} {Δ : KEnv ℓ₁} {κ : Kind ℓ₂}
      → TVar (Δ , κ) κ
  S : ∀ {ℓ₁ ℓ₂ ℓ₃} {Δ : KEnv ℓ₁} {κ : Kind ℓ₂} {κ' : Kind ℓ₃}
      → TVar Δ κ → TVar (Δ , κ') κ

--------------------------------------------------------------------------------
-- Types.

data Ty where
  ------------------------------------------------------------
  -- Base types (for mechanization).

  -- Unit (Mechanization.)
  U : ∀ {ℓ : Level} {Δ : KEnv ℓ} →

         --------------
         Ty Δ ★₀
  ------------------------------------------------------------
  -- System Fω.

  tvar : ∀ {ℓ₁ ℓ₂ : Level} {Δ : KEnv ℓ₁} {κ : Kind ℓ₂} →

         TVar Δ κ →
         -----------
         Ty Δ κ

  _`→_ : ∀ {ℓ₁ ℓ₂ ℓ₃ : Level} {Δ : KEnv ℓ₁} →
          Ty Δ (★ ℓ₂) → Ty Δ (★ ℓ₃) →
          -----------------------------------
          Ty Δ (★ (ℓ₂ ⊔ ℓ₃))

  `∀ :  ∀ {ℓ₁ ℓ₂ ℓ₃ : Level} {Δ : KEnv ℓ₁} →
          (κ : Kind ℓ₃) → Ty (Δ , κ) (★ ℓ₂) →
          -------------------------------------
          Ty Δ (★ (ℓ₂ ⊔ (lsuc ℓ₃)))

  `λ :  ∀ {ℓ₁ ℓ₂ ℓ₃ : Level} {Δ : KEnv ℓ₁} (κ₁ : Kind ℓ₂) {κ₂ : Kind ℓ₃} →
          Ty (Δ , κ₁) κ₂ →
          -----------------------------------------
          Ty Δ (κ₁ `→ κ₂)

  _·[_] : ∀ {ℓ₁ ℓ₂ ℓ₃ : Level} {Δ : KEnv ℓ₁} {κ₁ : Kind ℓ₂} {κ₂ : Kind ℓ₃} →
          Ty Δ (κ₁ `→ κ₂) → Ty Δ κ₁ →
          -----------------------------
          Ty Δ κ₂
  ------------------------------------------------------------
  -- Qualified types.

  _⇒_ : ∀ {ℓ ℓκ ℓτ} {κ : Kind ℓκ} {Δ : KEnv ℓ}
          → (π : Pred Δ κ) → Ty Δ (★ ℓτ) →
         --------------------------------
         Ty Δ (★ (lsuc ℓκ ⊔ ℓτ))

  ------------------------------------------------------------
  -- System Rω.

  -- Labels.
  lab :  ∀ {ℓ : Level} {Δ : KEnv ℓ} →
         Label →
         ----------
         Ty {ℓ}{lzero} Δ L

  -- singleton formation.
  _▹_ : ∀ {ℓ ι : Level} {Δ : KEnv ℓ} {κ : Kind ι} →
         Ty Δ L → Ty Δ κ →
         -------------------
         Ty Δ κ

  -- Row singleton formation.
  _R▹_ : ∀ {ℓ ι : Level} {Δ : KEnv ℓ} {κ : Kind ι} →
         Ty Δ L → Ty Δ κ →
         -------------------
         Ty Δ R[ κ ]

  -- label constant formation.
  ⌊_⌋ : ∀ {ℓ ι : Level} {Δ : KEnv ℓ} →
         Ty Δ L →
         ----------
         Ty Δ (★ ι)

  -- The empty record (mechanization only.)
  ∅ : ∀ {ℓ ι : Level} {Δ : KEnv ℓ} →
  
      --------------
      Ty Δ (★ ι)

  -- Record formation.
  Π : ∀ {ℓ ι : Level} {Δ : KEnv ℓ} →
      Ty Δ R[ ★ ι ] →
      -------------
      Ty Δ (★ ι)

  -- Variant formation.
  Σ : ∀ {ℓ ι : Level} {Δ : KEnv ℓ} →
      Ty Δ R[ ★ ι ] →
      -------------
      Ty Δ (★ ι)

  -- lift₁ (lifting a function argument to row kind).
  _·⌈_⌉ : ∀ {ℓ ι} {Δ : KEnv ℓ}
            {κ₁ κ₂ : Kind ι} →
          Ty Δ R[ κ₁ `→ κ₂ ] → Ty Δ κ₁ →
          --------------------------------
          Ty Δ R[ κ₂ ]

  -- lift₂ (lifting a function to row kind.)
  ⌈_⌉·_ : ∀ {ℓ ι} {Δ : KEnv ℓ}
            {κ₁ κ₂ : Kind ι} →
          Ty Δ (κ₁ `→ κ₂) → Ty Δ R[ κ₁ ] →
          --------------------------------
          Ty Δ R[ κ₂ ]
